"""
Live monkey/boar detector.
Grabs frames from your phone's IP Webcam feed and sends them to Roboflow
for detection, printing results as it goes.

Before running:
    pip install opencv-python requests

Usage:
    python live_detect.py
"""

import cv2
import requests
import base64
import time

# ---- SETTINGS: edit these ----
CAMERA_URL = "http://192.168.1.181:8080/video"   # your phone's IP Webcam address
API_KEY = "rBelYlcd54HgrRJoeoez"

MODEL_URL = "https://detect.roboflow.com/wild-animal-detection-sbzdi/6"

# How often to send a frame to Roboflow, in seconds (don't send every single
# frame — that would be very slow and could exceed free-tier limits)
CHECK_INTERVAL = 2
# --------------------------------


def send_to_roboflow(frame):
    """Encode a frame and send it to Roboflow, return the JSON result."""
    _, buffer = cv2.imencode(".jpg", frame)
    img_base64 = base64.b64encode(buffer).decode("utf-8")

    response = requests.post(
        MODEL_URL,
        params={"api_key": API_KEY},
        data=img_base64,
        headers={"Content-Type": "application/x-www-form-urlencoded"},
    )
    return response.json()


def main():
    cap = cv2.VideoCapture(CAMERA_URL)
    if not cap.isOpened():
        print("Could not open camera stream. Check the CAMERA_URL and WiFi connection.")
        return

    last_check = 0

    while True:
        ret, frame = cap.read()
        if not ret:
            print("Lost connection to camera.")
            break

        cv2.imshow("Live Feed", frame)

        # Only send a frame to Roboflow every CHECK_INTERVAL seconds
        now = time.time()
        if now - last_check >= CHECK_INTERVAL:
            last_check = now
            try:
                result = send_to_roboflow(frame)
                predictions = result.get("predictions", [])
                if predictions:
                    for p in predictions:
                        print(
                            f"[ALERT] {p['class']} detected! "
                            f"confidence={p['confidence']:.2f}"
                        )
                else:
                    print("No detection.")
            except Exception as e:
                print("Error contacting Roboflow:", e)

        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()


if __name__ == "__main__":
    main()
